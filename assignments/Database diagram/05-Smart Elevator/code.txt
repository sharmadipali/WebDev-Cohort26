  building_id serial pk

  location text
  created_at timestamp
  updated_at timestamp
}

  floor_id serial pk
  building_id int fk unique

  created_at timestamp
  updated_at timestamp
}

  shaft_id serial pk
  building_id int fk unique
  shaft_id int fk unique
  start_floor INT NOT NULL,
    end_floor INT NOT NULL,
  shaft_label varchar 
  travel_distance float
  is_active boolean
  created_at timestamp
  updated_at timestamp
}
Elevators[color: Yellow] {
  elevator_id serial pk
  shaft_id int fk
  model_number varchar
  capacity int
  current_floor int
 created_at timestamp
 updated_at timestamp
}

  direction varchar(10)

assigned_at timestamp

 created_at timestamp
  updated_at timestamp

 created_at timestamp
  updated_at timestamp

 created_at timestamp
